How Live Sync Works

1
Create a Room Open the dashboard on your primary device and create a sync room. You'll get a unique code.
2
Join from Other Devices Open the dashboard on your phone/laptop and enter the room code to connect.
3
Every Change Auto-Syncs Add a visit, edit a note, or save training it auto-pushes to all connected devices in real-time. No manual sync needed!
4
Auto-Reconnect If you refresh the page or lose connection, it automatically reconnects using the saved room code like WhatsApp Web.
Data travels directly between your devices (peer-to-peer). Nothing is stored on any server. All data is encrypted in transit.
Smart Merge: When both devices have changes, records are intelligently merged by ID no data loss. Newer records take priority.

function doPost(e) {
 var content = JSON.parse(e.postData.contents);
 var action = content.action;

 if (action === 'ping') {
 return send({status:'ok', app:'APF Dashboard Backup', version:2});
 }

 // === Per-Module Backup (v2) ===
 if (action === 'backup_modules') {
 var folder = getOrCreateFolder();
 var modules = content.modules || {};
 var saved = [];
 for (var key in modules) {
 var fname = 'APF_' + key + '.apf';
 var files = folder.getFilesByName(fname);
 if (files.hasNext()) {
 files.next().setContent(modules[key]);
 } else {
 folder.createFile(fname, modules[key], 'text/plain');
 }
 saved.push(key);
 }
 // Save manifest with timestamp
 var manifest = {timestamp:content.timestamp, modules:saved};
 var mFiles = folder.getFilesByName('APF_manifest.json');
 if (mFiles.hasNext()) {
 mFiles.next().setContent(JSON.stringify(manifest));
 } else {
 folder.createFile('APF_manifest.json', JSON.stringify(manifest), 'application/json');
 }
 return send({success:true, saved:saved, timestamp:content.timestamp});
 }

 if (action === 'restore_modules') {
 var folder = getOrCreateFolder();
 var result = {};
 var allFiles = folder.getFiles();
 while (allFiles.hasNext()) {
 var f = allFiles.next();
 var n = f.getName();
 if (n.indexOf('APF_') === 0 && n.indexOf('.apf') > 0 && n !== 'APF_Backup.apf') {
 var modKey = n.replace('APF_','').replace('.apf','');
 result[modKey] = f.getBlob().getDataAsString();
 }
 }
 if (Object.keys(result).length > 0) {
 return send({version:2, modules:result});
 }
 // Fallback: try legacy single file
 var legacy = folder.getFilesByName('APF_Backup.apf');
 if (legacy.hasNext()) {
 return send({version:1, data:legacy.next().getBlob().getDataAsString()});
 }
 return send({error:'No backup found'});
 }

 if (action === 'info_modules') {
 var folder = getOrCreateFolder();
 var mods = {};
 var totalSize = 0;
 var allFiles = folder.getFiles();
 while (allFiles.hasNext()) {
 var f = allFiles.next();
 var n = f.getName();
 if (n.indexOf('APF_') === 0 && n.indexOf('.apf') > 0 && n !== 'APF_Backup.apf') {
 var k = n.replace('APF_','').replace('.apf','');
 var sz = f.getSize();
 mods[k] = {size:sz, updated:f.getLastUpdated().toISOString()};
 totalSize += sz;
 }
 }
 // Check manifest
 var mFiles = folder.getFilesByName('APF_manifest.json');
 var manifest = null;
 if (mFiles.hasNext()) {
 try { manifest = JSON.parse(mFiles.next().getBlob().getDataAsString()); } catch(e) {}
 }
 return send({exists:Object.keys(mods).length>0, modules:mods, totalSize:totalSize, manifest:manifest});
 }

 // === Legacy single-file actions (backward compat) ===
 if (action === 'backup') {
 var folder = getOrCreateFolder();
 var files = folder.getFilesByName('APF_Backup.apf');
 if (files.hasNext()) {
 files.next().setContent(content.data);
 } else {
 folder.createFile('APF_Backup.apf', content.data, 'text/plain');
 }
 return send({success:true, timestamp:content.timestamp});
 }

 if (action === 'restore') {
 var folder = getOrCreateFolder();
 var files = folder.getFilesByName('APF_Backup.apf');
 if (files.hasNext()) {
 return send({data:files.next().getBlob().getDataAsString()});
 }
 return send({error:'No backup found'});
 }

 if (action === 'info') {
 var folder = getOrCreateFolder();
 var files = folder.getFilesByName('APF_Backup.apf');
 if (files.hasNext()) {
 var f = files.next();
 return send({exists:true, size:f.getSize(),
 updated:f.getLastUpdated().toISOString()});
 }
 return send({exists:false});
 }

 return send({error:'Unknown action'});
}

function doGet(e) {
 return ContentService.createTextOutput(
 JSON.stringify({status:'ok', app:'APF Dashboard Backup'})
 ).setMimeType(ContentService.MimeType.JSON);
}

function send(obj) {
 return ContentService.createTextOutput(
 JSON.stringify(obj)
 ).setMimeType(ContentService.MimeType.JSON);
}

function getOrCreateFolder() {
 var folders = DriveApp.getFoldersByName('APF Dashboard Backups');
 if (folders.hasNext()) return folders.next();
 return DriveApp.createFolder('APF Dashboard Backups');
}
